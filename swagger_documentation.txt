  /api_registrar_cuestionario:
    post:
      tags:
        - Cuestionario
      summary: Registrar un nuevo cuestionario
      description: Crea un nuevo cuestionario en el sistema. Genera un PIN aleatorio automáticamente.
      operationId: registrarCuestionario
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              required:
                - nombre
                - id_docente
              properties:

                  example: 1
                  description: ID del docente que crea el cuestionario
                estado:
                  type: string
                  enum: [P, R, Público, Privado]
                  default: P
                  example: "P"
                  description: Estado del cuestionario (P=Público, R=Privado)
                tipo_cuestionario:
                  type: string
                  enum: [I, G]
                  default: I
                  example: "I"
                  description: Tipo de cuestionario (I=Individual, G=Grupal)
                descripcion:
                  type: string
                  example: "Cuestionario sobre álgebra básica"
                  description: Descripción del cuestionario (opcional)
      responses:
        '201':
          description: Cuestionario registrado correctamente
          content:
            application/json:
              schema:
                allOf:
                  - $ref: '#/components/schemas/SuccessResponse'
                  - type: object
                    properties:
                      data:
                        type: array
                        items:
                          type: object
                          properties:
                            id_cuestionario:
                              type: integer
                              example: 1
                            pin:
                              type: string
                              example: "12345"
                      message:
                        type: string
                        example: "Cuestionario registrado correctamente"
        '400':
          description: Error de validación
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
              examples:
                camposFaltantes:
                  value:
                    data: []
                    message: "Faltan campos obligatorios: nombre, id_docente"
                    status: 0
                nombreCorto:
                  value:
                    data: []
                    message: "El nombre del cuestionario debe tener al menos 3 caracteres"
                    status: 0
                nombreLargo:
                  value:
                    data: []
                    message: "El nombre del cuestionario no puede exceder 200 caracteres"
                    status: 0
        '500':
          description: Error del servidor
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
